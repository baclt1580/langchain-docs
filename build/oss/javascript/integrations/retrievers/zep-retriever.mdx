> [Zep](https://www.getzep.com) 是一个面向 AI 助手应用的长时记忆服务。
> 通过 Zep，您可以让 AI 助手具备回忆过往对话的能力，无论对话发生在多久以前，
> 同时还能减少幻觉、降低延迟和成本。

> 对 Zep Cloud 感兴趣？请参阅 [Zep Cloud 安装指南](https://help.getzep.com/sdks)

本示例展示了如何在检索链中使用 Zep 检索器，从 Zep 开源记忆存储中检索文档。

## 安装

按照 [Zep 开源快速入门指南](https://help.getzep.com/quickstart) 安装并开始使用 Zep。

## 设置

<Tip>
有关安装 LangChain 包的通用说明，请参阅 [此部分](/oss/javascript/langchain/install)。
</Tip>

```bash npm
npm i @getzep/zep-js @langchain/community @langchain/core
```

## 使用

```typescript
import { ZepRetriever } from "@langchain/community/retrievers/zep";
import { ZepMemory } from "@langchain/community/memory/zep";
import { Memory as MemoryModel, Message } from "@getzep/zep-js";
import { randomUUID } from "crypto";

function sleep(ms: number) {
  // eslint-disable-next-line no-promise-executor-return
  return new Promise((resolve) => setTimeout(resolve, ms));
}

export const run = async () => {
  const zepConfig = {
    url: process.env.ZEP_URL || "http://localhost:8000",
    sessionId: `session_${randomUUID()}`,
  };

  console.log(`Zep 配置: ${JSON.stringify(zepConfig)}`);

  const memory = new ZepMemory({
    baseURL: zepConfig.url,
    sessionId: zepConfig.sessionId,
  });

  // 生成关于法国旅行的聊天消息
  const chatMessages = [
    {
      role: "AI",
      message: "Bonjour！今天我能如何协助您的旅行计划？",
    },
    { role: "User", message: "我正在计划去法国旅行。" },
    {
      role: "AI",
      message: "听起来很令人兴奋！您计划访问哪些城市？",
    },
    { role: "User", message: "我考虑去巴黎和尼斯。" },
    {
      role: "AI",
      message: "很棒的选择！您对任何特定活动感兴趣吗？",
    },
    { role: "User", message: "我很想去参观一些葡萄园。" },
    {
      role: "AI",
      message:
        "法国拥有世界上一些最好的葡萄园。我可以帮您找一些。",
    },
    { role: "User", message: "那太好了！" },
    { role: "AI", message: "您更喜欢红葡萄酒还是白葡萄酒？" },
    { role: "User", message: "我更喜欢红葡萄酒。" },
    {
      role: "AI",
      message:
        "完美！我会找一些以红葡萄酒闻名的葡萄园。",
    },
    { role: "User", message: "谢谢，那会非常有帮助。" },
    {
      role: "AI",
      message:
        "不客气！我还会为您查找一些法国葡萄酒礼仪。",
    },
    {
      role: "User",
      message: "听起来很棒。我迫不及待想开始我的旅行了！",
    },
    {
      role: "AI",
      message:
        "我相信您会度过一段美好的时光。关于您的旅行还有其他问题吗？",
    },
    { role: "User", message: "暂时没有，谢谢您的帮助！" },
  ];

  const zepClient = await memory.zepClientPromise;
  if (!zepClient) {
    throw new Error("ZepClient 未初始化");
  }

  // 将聊天消息添加到记忆
  for (const chatMessage of chatMessages) {
    let m: MemoryModel;
    if (chatMessage.role === "AI") {
      m = new MemoryModel({
        messages: [new Message({ role: "ai", content: chatMessage.message })],
      });
    } else {
      m = new MemoryModel({
        messages: [
          new Message({ role: "human", content: chatMessage.message }),
        ],
      });
    }

    await zepClient.memory.addMemory(zepConfig.sessionId, m);
  }

  // 等待消息被总结、丰富、嵌入和索引。
  await sleep(10000);

  // 简单相似性搜索
  const query = "我可以在法国开红色的车吗？";
  const retriever = new ZepRetriever({ ...zepConfig, topK: 3 });
  const docs = await retriever.invoke(query);
  console.log("简单相似性搜索");
  console.log(JSON.stringify(docs, null, 2));

  // MMR 重排序搜索
  const mmrRetriever = new ZepRetriever({
    ...zepConfig,
    topK: 3,
    searchType: "mmr",
    mmrLambda: 0.5,
  });
  const mmrDocs = await mmrRetriever.invoke(query);
  console.log("MMR 重排序搜索");
  console.log(JSON.stringify(mmrDocs, null, 2));

  // 摘要搜索配合 MMR 重排序
  const mmrSummaryRetriever = new ZepRetriever({
    ...zepConfig,
    topK: 3,
    searchScope: "summary",
    searchType: "mmr",
    mmrLambda: 0.5,
  });
  const mmrSummaryDocs = await mmrSummaryRetriever.invoke(query);
  console.log("摘要搜索配合 MMR 重排序");
  console.log(JSON.stringify(mmrSummaryDocs, null, 2));

  // 过滤搜索
  const filteredRetriever = new ZepRetriever({
    ...zepConfig,
    topK: 3,
    filter: {
      where: { jsonpath: '$.system.entities[*] ? (@.Label == "GPE")' },
    },
  });
  const filteredDocs = await filteredRetriever.invoke(query);
  console.log("过滤搜索");
  console.log(JSON.stringify(filteredDocs, null, 2));
};
```
